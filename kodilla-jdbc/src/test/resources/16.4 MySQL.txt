SELECT I.SUMMARY AS ISSUE, I.DESCRIPTION, L.NAME AS ISSUE_LIST
FROM issues I JOIN issueslists L ON I.ISSUESLIST_ID = L.ID
;

SELECT I.SUMMARY AS ISSUE,I.DESCRIPTION,U.LASTNAME, U.FIRSTNAME
FROM issues I JOIN users u on I.USER_ID_ASSIGNEDTO = u.ID
;

SELECT U.FIRSTNAME,U.LASTNAME, COUNT(*) AS NUMBER_OF_ISSUES
FROM users U JOIN issues I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*)>1
;
